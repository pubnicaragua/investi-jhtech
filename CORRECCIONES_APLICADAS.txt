========================================
CÓMO FUNCIONA AHORA:
========================================

1. Carga inicial: 20 posts de la base de datos
2. Al hacer scroll: Repite los posts originales
3. IDs únicos: Usa timestamp + random para evitar warnings
4. Infinito real: NUNCA se detiene el scroll

========================================
CÓDIGO APLICADO:
========================================

const loadMorePosts = async () => {
  if (loadingMore || !userId) return
  
  setLoadingMore(true)
  try {
    // Obtener los posts originales (sin duplicados)
    const originalPosts = posts.filter(p => !p.id.includes('-'))
    
    // Si no hay posts originales, usar todos los posts
    const postsToRepeat = originalPosts.length > 0 ? originalPosts : posts.slice(0, 20)
    
    // Crear nuevos posts con IDs únicos
    const morePosts = postsToRepeat.map(post => ({
      ...post,
      id: `${post.id.split('-')[0]}-repeat-${Date.now()}-${Math.random()}`
    }))
    
    setPosts([...posts, ...morePosts])
    setPage(page + 1)
    setHasMore(true) // SIEMPRE true
  } catch (err) {
    console.error("Error loading more posts:", err)
  } finally {
    setLoadingMore(false)
  }
}
